### 2. 操作数候选池构建

**数据结构操作：**
```python
def build_operand_candidates(available_tables, column_usage_tracker):
    # 创建类型分组字典 - 操作字典数据结构
    candidates = {
        'numeric': [],      # 数值类型操作数
        'string': [],       # 字符串类型操作数  
        'datetime': []      # 日期时间类型操作数
    }
    
    # 遍历所有表和列
    for table in available_tables:
        table_alias = column_usage_tracker.get_table_alias(table)
        for column in table.columns:
            # 操作used_columns集合 - 检查列是否已使用
            if (table_alias, column.name) not in column_usage_tracker.used_columns:
                # 根据列类型分类 - 操作类型集合
                if column.data_type in numeric_types:
                    type_group = 'numeric'
                elif column.data_type in string_types:
                    type_group = 'string' 
                elif column.data_type in datetime_types:
                    type_group = 'datetime'
                else:
                    continue
                
                # 创建列引用并添加到对应类型组 - 操作列表
                column_ref = ColumnReferenceNode(column, table_alias)
                candidates[type_group].append(column_ref)
    
    return candidates
```

### 3. 左操作数选择流程

**数据结构操作：**
```python
def select_left_operand(candidates, available_tables, column_usage_tracker):
    # 获取非空类型组 - 操作字典和列表
    non_empty_groups = [group for group, group_candidates in candidates.items() if group_candidates]
    
    if not non_empty_groups:
        return None, None
    
    # 应用类型优先级 - 操作列表
    preferred_groups = ['numeric', 'string', 'datetime']
    selected_group = None
    
    for group in preferred_groups:
        if group in non_empty_groups:
            selected_group = group
            break
    
    if not selected_group:
        selected_group = non_empty_groups[0]
    
    # 获取所选类型组的候选列 - 操作字典
    group_candidates = candidates[selected_group]
    
    # 优先选择索引列 - 操作列表过滤
    indexed_candidates = [cand for cand in group_candidates if cand.column.is_indexed]
    
    if indexed_candidates:
        left_operand = random.choice(indexed_candidates)
    else:
        left_operand = random.choice(group_candidates)
    
    # 标记左操作数为已使用 - 操作used_columns集合
    column_usage_tracker.mark_column_as_used(left_operand)
    
    return left_operand, selected_group
```

### 4. 右操作数选择与类型匹配

**数据结构操作：**
```python
def select_right_operand(left_operand, left_type_group, candidates, available_tables):
    # 获取同类型候选 - 操作字典
    same_type_candidates = candidates.get(left_type_group, [])
    
    # 根据类型组选择右操作数
    if left_type_group == 'numeric':
        # 50%概率选择列引用，50%概率选择字面量
        if same_type_candidates and random.random() > 0.5:
            right_operand = random.choice(same_type_candidates)
        else:
            # 生成数值字面量
            value = random.randint(-1000, 1000)
            right_operand = LiteralNode(value, 'INT')
            
    elif left_type_group == 'string':
        # 40%概率选择列引用，60%概率选择字面量
        if same_type_candidates and random.random() > 0.6:
            right_operand = random.choice(same_type_candidates)
        else:
            # 生成字符串字面量
            string_patterns = ['test_value_', 'sample_', 'data_', 'prefix_', 'suffix_']
            base_pattern = random.choice(string_patterns)
            string_value = base_pattern + str(random.randint(1, 100))
            right_operand = LiteralNode(string_value, 'VARCHAR')
            
    elif left_type_group == 'datetime':
        # 30%概率选择列引用，70%概率选择字面量
        if same_type_candidates and random.random() > 0.7:
            right_operand = random.choice(same_type_candidates)
        else:
            # 生成日期字面量
            year = random.randint(2000, 2025)
            month = random.randint(1, 12)
            day = random.randint(1, 28)
            date_value = f"{year:04d}-{month:02d}-{day:02d}"
            right_operand = LiteralNode(date_value, 'DATE')
    
    return right_operand
```

### 5. 类型兼容性检查

**数据结构操作：**
```python
def is_type_compatible(left_type, right_type):
    """检查两个数据类型是否兼容"""
    # 提取基本类型 - 字符串操作
    base_type1 = left_type.split('(')[0].upper() if left_type else 'UNKNOWN'
    base_type2 = right_type.split('(')[0].upper() if right_type else 'UNKNOWN'
    
    # 使用类型集合进行兼容性检查
    if (base_type1 in numeric_types and base_type2 in numeric_types) or \
       (base_type1 in string_types and base_type2 in string_types) or \
       (base_type1 in datetime_types and base_type2 in datetime_types) or \
       base_type1 == base_type2:
        return True
    
    return False
```

## 类型修复机制

### 1. 类型修复流程

**数据结构操作：**
```python
def repair_comparison_types(comparison_node, dialect_config):
    """修复比较表达式中的类型不兼容问题"""
    if len(comparison_node.children) < 2:
        return comparison_node
    
    left_operand = comparison_node.children[0]
    right_operand = comparison_node.children[1]
    
    left_type = get_node_type(left_operand)
    right_type = get_node_type(right_operand)
    
    # 检查类型兼容性 - 操作类型集合
    if not is_type_compatible(left_type, right_type):
        # 提取基本类型
        base_left = left_type.split('(')[0].upper() if left_type else ''
        base_right = right_type.split('(')[0].upper() if right_type else ''
        
        # 数值与字符串的转换
        if base_left in numeric_types and base_right in string_types:
            # 构建CAST函数调用 - 操作FunctionCallNode的children列表
            cast_func = Function('CAST', base_left, ['string'])
            cast_node = FunctionCallNode(cast_func)
            cast_node.children.append(right_operand)
            
            # 替换右操作数 - 操作ComparisonNode的children列表
            comparison_node.children[1] = cast_node
            
        elif base_left in string_types and base_right in numeric_types:
            # 构建CAST函数调用
            cast_func = Function('CAST', base_right, ['string'])
            cast_node = FunctionCallNode(cast_func)
            cast_node.children.append(left_operand)
            
            # 替换左操作数
            comparison_node.children[0] = cast_node
    
    return comparison_node
```

### 2. 安全算术表达式生成

```python
def generate_safe_arithmetic_expression(left_operand, right_operand, operator):
    """生成安全的算术表达式，包含除零保护"""
    arithmetic_node = ArithmeticNode(operator)
    arithmetic_node.children.append(left_operand)  # 操作children列表
    
    # 对于除法和取模运算，添加除零保护
    if operator in ['/', '%']:
        # 使用NULLIF函数进行除零保护
        nullif_func = Function('NULLIF', 'numeric', ['numeric', 'numeric'])
        nullif_node = FunctionCallNode(nullif_func)
        nullif_node.children.append(right_operand)        # 操作children列表
        nullif_node.children.append(LiteralNode(0, 'INT')) # 操作children列表
        
        arithmetic_node.children.append(nullif_node)      # 操作children列表
    else:
        arithmetic_node.children.append(right_operand)    # 操作children列表
    
    return arithmetic_node
```

## 关键数据结构操作总结

| 流程步骤 | 操作的数据结构 | 具体操作 | 目的 | 时间复杂度 |
|---------|---------------|----------|------|------------|
| 候选池构建 | 类型分组字典 | 按类型分组列引用 | 组织候选操作数 | O(n) |
| 类型检查 | 类型集合 | in 操作符检查 | 快速类型分类 | O(1) |
| 左操作数选择 | used_columns 集合 | add((alias, name)) | 标记列已使用 | O(1) |
| 操作符映射 | 优先级字典 | get(operator) | 获取类型优先级 | O(1) |
| 表达式构建 | children 列表 | append() 操作数 | 构建表达式结构 | O(1) |
| 类型修复 | children 列表 | 索引替换元素 | 修复类型不兼容 | O(1) |
| 安全增强 | FunctionCallNode | 构建嵌套结构 | 添加安全保护 | O(1) |

## 算法优势

1. **类型安全性**：通过严格的类型检查和修复机制确保表达式类型安全
2. **语义正确性**：根据操作符特性选择合适的操作数类型
3. **性能优化**：优先选择索引列，提高查询性能
4. **多样性生成**：平衡使用列引用和字面量，增加表达式多样性
5. **跨方言兼容**：支持不同数据库方言的类型转换策略
6. **错误恢复**：具备类型修复能力，提高算法鲁棒性

## 结论

表达式类型兼容性算法通过精心设计的数据结构组合，实现了高效、安全的表达式生成和管理。类型集合提供快速的类型分类，字典结构维护操作符到类型优先级的映射，列表结构管理表达式操作数，共同协作确保生成的表达式类型正确且语义合理。

该算法不仅支持基本的比较和算术表达式，还能处理复杂的类型转换、安全保护和特殊操作符需求。通过与列使用跟踪器的紧密集成，确保了表达式生成的列引用有效性和查询性能优化。类型修复机制进一步提高了算法的适应性，使其能够处理各种类型不兼容的情况。

这种设计模式为SQL查询生成提供了强大的表达式支持，为生成复杂、多样且类型安全的SQL查询奠定了坚实基础。
